var dup = function(words, rep) {
    var rep = rep || 5;
    var ret = [];
    for (var i = 0; i < words.length; ++i) {
        for (var j = 0; j < rep; ++j) {
            ret.push(words[i]);
        }
    }
    return ret;
}
